## Task 1: llm-stream-parser

**Files:**
- Create: `lib/agent/llm-stream-parser.ts`
- Create: `lib/agent/tests/llm-stream-parser.test.ts`

**Context:** LLM 流式输出形如：

```
<thinking>
这是思考内容，可能很长很多行...
</thinking>
<action>{"type":"raise","amount":80}</action>
```

parser 需要：
1. 边接收边检测 `<thinking>` 起始，推入 "thinking chunk" 给 SSE 使用
2. 检测 `</thinking>` 结束，close 思考通道
3. 检测 `<action>...</action>`，解析 JSON 返回 PokerAction
4. 支持宽松匹配：大小写不敏感、允许 `</ thinking>`、允许缺失闭标签（用 `</action>` 的出现收尾）

- [x] **Step 1: 类型**

```typescript
// lib/agent/llm-stream-parser.ts
export type ParserEvent =
  | { kind: 'thinking_delta'; text: string }
  | { kind: 'thinking_end' }
  | { kind: 'action'; action: unknown }
  | { kind: 'raw'; text: string }      // 未进 thinking/action 区块的原始文本

export class LlmStreamParser {
  private buffer = ''
  private inThinking = false
  private sawAction = false

  feed(chunk: string): ParserEvent[] {
    this.buffer += chunk
    const events: ParserEvent[] = []

    while (true) {
      if (!this.inThinking && !this.sawAction) {
        const startIdx = this.indexOfCI('<thinking>')
        const actStartIdx = this.indexOfCI('<action>')
        if (startIdx !== -1 && (actStartIdx === -1 || startIdx < actStartIdx)) {
          this.buffer = this.buffer.slice(startIdx + '<thinking>'.length)
          this.inThinking = true
          continue
        }
        if (actStartIdx !== -1) {
          this.buffer = this.buffer.slice(actStartIdx + '<action>'.length)
          this.sawAction = true
          continue
        }
        // 没有标签，等更多
        break
      }

      if (this.inThinking) {
        const endIdx = this.indexOfCI('</thinking>')
        if (endIdx === -1) {
          // 尽量 flush 除最后 12 字符（避免半标签）的内容
          const safeLen = Math.max(0, this.buffer.length - 12)
          if (safeLen > 0) {
            events.push({ kind: 'thinking_delta', text: this.buffer.slice(0, safeLen) })
            this.buffer = this.buffer.slice(safeLen)
          }
          break
        } else {
          if (endIdx > 0) {
            events.push({ kind: 'thinking_delta', text: this.buffer.slice(0, endIdx) })
          }
          events.push({ kind: 'thinking_end' })
          this.buffer = this.buffer.slice(endIdx + '</thinking>'.length)
          this.inThinking = false
          continue
        }
      }

      if (this.sawAction) {
        const endIdx = this.indexOfCI('</action>')
        if (endIdx === -1) break
        const body = this.buffer.slice(0, endIdx).trim()
        try {
          events.push({ kind: 'action', action: JSON.parse(body) })
        } catch {
          events.push({ kind: 'action', action: { error: 'json_parse', raw: body } })
        }
        this.buffer = this.buffer.slice(endIdx + '</action>'.length)
        this.sawAction = false
        continue
      }
    }
    return events
  }

  end(): ParserEvent[] {
    // 结束时 flush 剩余
    const events: ParserEvent[] = []
    if (this.inThinking && this.buffer.length > 0) {
      events.push({ kind: 'thinking_delta', text: this.buffer })
      events.push({ kind: 'thinking_end' })
    }
    if (this.sawAction && this.buffer.length > 0) {
      const body = this.buffer.trim()
      try {
        events.push({ kind: 'action', action: JSON.parse(body) })
      } catch {
        events.push({ kind: 'action', action: { error: 'json_parse', raw: body } })
      }
    }
    this.buffer = ''
    return events
  }

  private indexOfCI(needle: string): number {
    return this.buffer.toLowerCase().indexOf(needle.toLowerCase())
  }
}
```

- [x] **Step 3: llm-runtime.ts**

```typescript
import { createOpenAICompatible } from '@ai-sdk/openai-compatible'
import { createAnthropic } from '@ai-sdk/anthropic'
import { streamText } from 'ai'
import { LlmStreamParser } from './llm-stream-parser'
import { LlmError } from './llm-errors'

export interface LlmRuntimeInput {
  profile: {
    providerKind: 'openai_compatible' | 'anthropic'
    baseUrl: string
    apiKey: string
    model: string
  }
  agent: {
    systemPrompt: string
  }
  userPrompt: string
  timeoutMs?: number
  abortSignal?: AbortSignal
  onThinkingDelta?: (text: string) => void
}

export interface LlmRuntimeResult {
  action: unknown
  thinkingText: string
  rawResponse: string
}

export async function runDecision(input: LlmRuntimeInput): Promise<LlmRuntimeResult> {
  const { profile, agent, userPrompt, timeoutMs = 60_000, abortSignal, onThinkingDelta } = input
  const parser = new LlmStreamParser()
  let thinkingText = ''
  let action: unknown = null
  let raw = ''

  const controller = new AbortController()
  const combinedSignal = abortSignal
    ? mergeSignals(abortSignal, controller.signal)
    : controller.signal
  const timer = setTimeout(() => controller.abort(), timeoutMs)

  try {
    const model = buildModel(profile)
    const result = streamText({
      model,
      system: agent.systemPrompt,
      prompt: userPrompt,
      abortSignal: combinedSignal,
    })

    for await (const delta of result.textStream) {
      raw += delta
      for (const ev of parser.feed(delta)) {
        if (ev.kind === 'thinking_delta') {
          thinkingText += ev.text
          onThinkingDelta?.(ev.text)
        } else if (ev.kind === 'action') {
          action = ev.action
        }
      }
    }
    for (const ev of parser.end()) {
      if (ev.kind === 'thinking_delta') {
        thinkingText += ev.text
        onThinkingDelta?.(ev.text)
      } else if (ev.kind === 'action') {
        action = ev.action
      }
    }

    if (!action) throw new LlmError('parse_fail', 'no <action> found', { raw })
    return { action, thinkingText, rawResponse: raw }
  } catch (e) {
    if (controller.signal.aborted && abortSignal?.aborted !== true) {
      throw new LlmError('timeout', `llm timeout after ${timeoutMs}ms`, e)
    }
    if (e instanceof LlmError) throw e
    throw new LlmError('api_error', (e as Error).message ?? 'llm api error', e)
  } finally {
    clearTimeout(timer)
  }
}

function buildModel(p: LlmRuntimeInput['profile']) {
  if (p.providerKind === 'anthropic') {
    const anthropic = createAnthropic({ apiKey: p.apiKey, baseURL: p.baseUrl })
    return anthropic(p.model)
  }
  const openai = createOpenAICompatible({ name: 'llm', baseURL: p.baseUrl, apiKey: p.apiKey })
  return openai(p.model)
}

function mergeSignals(a: AbortSignal, b: AbortSignal): AbortSignal {
  if (a.aborted) return a
  if (b.aborted) return b
  const c = new AbortController()
  a.addEventListener('abort', () => c.abort())
  b.addEventListener('abort', () => c.abort())
  return c.signal
}
```

// lib/agent/key-cache.ts
const cache = new Map<string, string>() // key=matchId:profileId -> apiKey

export function putApiKey(matchId: string, profileId: string, apiKey: string) {
  cache.set(`${matchId}:${profileId}`, apiKey)
}

export function getApiKey(matchId: string, profileId: string): string | undefined {
  return cache.get(`${matchId}:${profileId}`)
}

export function dropMatch(matchId: string) {
  for (const k of cache.keys()) {
    if (k.startsWith(`${matchId}:`)) cache.delete(k)
  }
}

- [x] **Step 2: 改 agent stream route**

替换 `app/api/agents/[agentId]/message/stream/route.ts` 核心流程（伪代码，实际粘贴完整版）：

```typescript
import { NextRequest } from 'next/server'
import { getAgent, getProfile } from '@/db/queries/agents'
import { runDecision } from '@/lib/agent/llm-runtime'
import { LlmError } from '@/lib/agent/llm-errors'
import { runBotDecision } from '@/games/poker/agent/bot'
import { getApiKey } from '@/lib/agent/key-cache'
import { logAgentError } from '@/db/queries/errors'
import { verifyMatchToken } from '@/lib/auth/match-token'

export async function POST(
  req: NextRequest,
  { params }: { params: Promise<{ agentId: string }> },
) {
  const { agentId } = await params
  const body = await req.json()
  const { matchToken, matchId, contextText, fallbackInput } = body

  if (!verifyMatchToken(matchToken, matchId)) {
    return new Response('unauthorized', { status: 401 })
  }

  const agent = await getAgent(agentId)
  if (!agent) return new Response('agent not found', { status: 404 })
  const profile = await getProfile(agent.profileId)
  if (!profile) return new Response('profile not found', { status: 404 })
  const apiKey = getApiKey(matchId, profile.id)
  if (!apiKey) return new Response('no api key', { status: 400 })

  const encoder = new TextEncoder()
  const stream = new ReadableStream({
    async start(controller) {
      const send = (kind: string, payload: unknown) => {
        controller.enqueue(encoder.encode(`event: ${kind}\ndata: ${JSON.stringify(payload)}\n\n`))
      }

      try {
        const res = await runDecision({
          profile: {
            providerKind: profile.providerKind as any,
            baseUrl: profile.baseUrl,
            apiKey,
            model: agent.model,
          },
          agent: { systemPrompt: agent.systemPrompt },
          userPrompt: contextText,
          timeoutMs: 60_000,
          onThinkingDelta: (text) => send('thinking_delta', { text }),
        })
        send('action', { action: res.action, thinkingText: res.thinkingText })
      } catch (err) {
        const kind = err instanceof LlmError ? err.kind : 'api_error'
        await logAgentError({
          matchId, agentId, kind,
          message: (err as Error).message,
          raw: (err as any).cause?.raw ?? null,
        })
        const action = runBotDecision(fallbackInput)
        send('action', { action, fallback: true, errorKind: kind })
      } finally {
        controller.close()
      }
    },
  })

  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
    },
  })
}
```
