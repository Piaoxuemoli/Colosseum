```typescript
import type { ActionSpec } from '@/lib/engine/contracts'
import type { BotStrategy } from '@/lib/core/registry'
import { log } from '@/lib/telemetry/logger'

export type ValidationResult<TAction> = {
  action: TAction
  layer: 'structured' | 'parse' | 'validate' | 'fallback'
}

/**
 * 三层校验（spec §7.3）。
 * 本函数合并第二+三层：parse 已经产出候选 action，这里做 engine 合法性检查 + fallback。
 * 第一层 structured-output 在 LLM 调用层处理（P1b 及以后）。
 */
export function coerceToValidAction<TAction>(
  candidate: TAction,
  validActions: ActionSpec<TAction>[],
  state: unknown,
  botStrategy: BotStrategy,
  meta: { matchId: string; agentId: string; layerIfPassed: 'parse' | 'validate' },
): ValidationResult<TAction> {
  const actionType = (candidate as unknown as { type?: string }).type
  const matched = validActions.find((v) => v.type === actionType)
  if (matched) {
    return { action: candidate, layer: meta.layerIfPassed }
  }
  log.warn('action-validator: invalid action, falling back to BotStrategy', {
    matchId: meta.matchId, agentId: meta.agentId, candidate: candidate as unknown,
  })
  const fallback = botStrategy.decide(state, validActions as unknown[]) as TAction
  return { action: fallback, layer: 'fallback' }
}
```

```typescript
import { redis } from '@/lib/redis/client'
import { keys } from '@/lib/redis/keys'
import { getGame } from '@/lib/core/registry'
import type { MatchConfig, GameType } from '@/lib/core/types'
import { defaultMatchConfig } from '@/lib/core/types'
import { newMatchToken } from '@/lib/core/ids'
import {
  createMatch, listParticipants, findMatchById, updateMatchStatus, finalizeMatchRow,
} from '@/lib/db/queries/matches'
import { appendEvents, nextSeq } from '@/lib/db/queries/events'
import { deleteWorkingMemory } from '@/lib/db/queries/memory'
import { log } from '@/lib/telemetry/logger'

export type CreateMatchInput = {
  gameType: GameType
  agentIds: string[]            // 座位顺序
  config?: Partial<MatchConfig>
  keyring?: Record<string, string>   // profileId → apiKey（可选；Bot-only 测试时不传）
  engineConfig?: Record<string, unknown>  // 游戏特定（poker: {smallBlind, bigBlind, startingChips, maxBetsPerStreet}）
}

export async function createAndStartMatch(input: CreateMatchInput): Promise<{
  matchId: string
  token: string
}> {
  const cfg: MatchConfig = { ...defaultMatchConfig(), ...(input.config ?? {}) }
  const { matchId } = await createMatch({
    gameType: input.gameType,
    config: cfg,
    participants: input.agentIds.map((id, idx) => ({ agentId: id, seatIndex: idx })),
  })

  // 引擎初始化
  const gameMod = getGame(input.gameType)
  const engineConfig = input.engineConfig ?? {
    smallBlind: 2, bigBlind: 4, startingChips: 200, maxBetsPerStreet: 4,
  }
  const initState = gameMod.engine.createInitialState(engineConfig, input.agentIds)

  // 存 Redis
  await redis.set(keys.matchState(matchId), JSON.stringify(initState), 'EX', 24 * 3600)
  const token = newMatchToken()
  await redis.set(keys.matchToken(matchId), token, 'EX', 24 * 3600)
  if (input.keyring && Object.keys(input.keyring).length > 0) {
    await redis.hset(keys.matchKeyring(matchId), input.keyring)
    await redis.expire(keys.matchKeyring(matchId), 2 * 3600)
  }

  // createInitialState 产生的初始事件（发牌、盲注等）需要 persist
  // poker-engine 的 createInitialState 目前不返回 events；spec 允许这样，
  // 因为初始状态本身 = "事件已经发生过"。events 写入留给第一次 applyAction
  // 后续：GM 进入 running 时可以 append 一个 'poker/match-start' 事件。
  const seq = await nextSeq(matchId)
  await appendEvents([
    {
      id: `evt_start_${matchId}`,
      matchId, gameType: input.gameType, seq,
      occurredAt: new Date().toISOString(),
      kind: `${input.gameType}/match-start`,
      actorAgentId: null,
      payload: { participants: input.agentIds },
      visibility: 'public', restrictedTo: null,
    },
  ])

  await updateMatchStatus(matchId, 'running')
  log.info('match created', { matchId, gameType: input.gameType })

  return { matchId, token }
}

export async function finalizeMatch(matchId: string): Promise<void> {
  const match = await findMatchById(matchId)
  if (!match) return

  const stateRaw = await redis.get(keys.matchState(matchId))
  if (!stateRaw) {
    await updateMatchStatus(matchId, 'errored')
    return
  }
  const state = JSON.parse(stateRaw)
  const gameMod = getGame(match.gameType as GameType)
  const result = gameMod.engine.finalize(state)

  await finalizeMatchRow({ matchId, winnerFaction: result.winnerFaction, result })

  // 清 Redis
  await redis.del(keys.matchState(matchId))
  await redis.del(keys.matchKeyring(matchId))
  await redis.del(keys.matchToken(matchId))

  // 清 working memory
  const participants = await listParticipants(matchId)
  for (const p of participants) {
    await deleteWorkingMemory(p.agentId, matchId)
  }

  log.info('match finalized', { matchId, winnerFaction: result.winnerFaction })
}
```

```typescript
import { redis } from '@/lib/redis/client'
import { keys } from '@/lib/redis/keys'
import { getGame } from '@/lib/core/registry'
import { findMatchById, listParticipants } from '@/lib/db/queries/matches'
import { appendEvents, nextSeq } from '@/lib/db/queries/events'
import { recordAgentError } from '@/lib/db/queries/errors'
import { coerceToValidAction } from './action-validator'
import { publishSse } from './sse-broadcast'
import { finalizeMatch } from './match-lifecycle'
import { log } from '@/lib/telemetry/logger'
import type { GameType } from '@/lib/core/types'

export type TickResult = { done: boolean }

/**
 * 推进一次 match。Bot-only 模式：直接调 BotStrategy（不走 A2A Endpoint）。
 * LLM 模式（后续 Phase）：改为 requestAgentDecision()。
 */
export async function tickMatch(matchId: string): Promise<TickResult> {
  // 锁
  const locked = await redis.set(keys.matchLock(matchId), '1', 'EX', 60, 'NX')
  if (!locked) {
    log.info('tick skipped: locked', { matchId })
    return { done: false }
  }

  try {
    const match = await findMatchById(matchId)
    if (!match || match.status !== 'running') return { done: true }

    const stateRaw = await redis.get(keys.matchState(matchId))
    if (!stateRaw) {
      log.error('tick: no state in redis', { matchId })
      return { done: true }
    }
    const state = JSON.parse(stateRaw)

    const gameMod = getGame(match.gameType as GameType)
    const actorId = gameMod.engine.currentActor(state)

    if (!actorId) {
      // 边界：match-end
      if (state.matchComplete) {
        await finalizeMatch(matchId)
        await publishSse(matchId, { kind: 'match-end', winnerAgentId: null })
        return { done: true }
      }
      // handComplete 但不是 matchComplete：简化处理——目前引擎没有自动"开下一手"
      // 本 Phase 到此停止（Bot-only 跑到一手结束就行）
      log.info('tick: hand complete, no next-hand logic in P1a', { matchId })
      await publishSse(matchId, { kind: 'match-end', winnerAgentId: null })
      await finalizeMatch(matchId)
      return { done: true }
    }

    // 本 Phase: Bot-only 直接调 strategy
    const valid = gameMod.engine.availableActions(state, actorId)
    const botDecision = gameMod.botStrategy.decide(state, valid as unknown[])

    // 三层校验
    const { action, layer } = coerceToValidAction(
      botDecision, valid as never, state,
      gameMod.botStrategy,
      { matchId, agentId: actorId, layerIfPassed: 'parse' },
    )
    if (layer === 'fallback') {
      await recordAgentError({
        matchId, agentId: actorId, layer: 'fallback',
        errorCode: 'bot-invalid-action',
        recoveryAction: action as unknown as Record<string, unknown>,
      })
    }

    // 推进引擎
    const { nextState, events } = gameMod.engine.applyAction(state, actorId, action)

    // 分配 seq + matchId 给 events
    let seq = await nextSeq(matchId)
    const finalEvents = events.map((e) => ({ ...e, matchId, seq: seq++ }))
    await appendEvents(finalEvents)

    // 广播 public events
    for (const e of finalEvents) {
      if (e.visibility === 'public') {
        await publishSse(matchId, { kind: 'event', event: e })
      }
    }

    // 保存 state
    await redis.set(keys.matchState(matchId), JSON.stringify(nextState), 'EX', 24 * 3600)

    // boundary 处理
    const boundary = gameMod.engine.boundary(state, nextState)
    if (boundary === 'match-end') {
      await finalizeMatch(matchId)
      await publishSse(matchId, { kind: 'match-end', winnerAgentId: null })
      return { done: true }
    }

    return { done: false }
  } finally {
    await redis.del(keys.matchLock(matchId))
  }
}

/** Self-triggered tick loop: 一直 tick 到 done=true。 */
export async function runMatchToCompletion(
  matchId: string,
  options?: { maxTicks?: number; intervalMs?: number },
): Promise<void> {
  const max = options?.maxTicks ?? 1000
  const interval = options?.intervalMs ?? 0
  for (let i = 0; i < max; i++) {
    const r = await tickMatch(matchId)
    if (r.done) return
    if (interval > 0) await new Promise((res) => setTimeout(res, interval))
  }
  log.warn('runMatchToCompletion: maxTicks reached', { matchId, maxTicks: max })
}
```

**设计说明**：`runMatchToCompletion` 是同步内联版本，用于 integration test 里跑完一局；真正生产版本的 self-fetch tick loop 在 Task 25 的 route handler 里。
